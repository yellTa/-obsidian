# 유지보수
 

```dataviewjs
const pages = dv.pages('"1_Projects/Burgerput/2_유지보수"')
  .where(p => p.file.name !== "2_유지보수");

dv.table(
  ["파일 이름", "생성 날짜", "수정 날짜", "Progress"],
  pages.map(p => {
    const progress = p.progress ? String(p.progress) : "";
    let fileName;

    if (progress.toLowerCase() === "unsolved") {
      fileName = `<a href="obsidian://open?vault=First&file=${encodeURIComponent(p.file.path)}" style="color: red;">${p.file.name}</a>`;
    } else if (progress.toLowerCase() === "ongoing") {
      fileName = `<a href="obsidian://open?vault=First&file=${encodeURIComponent(p.file.path)}" style="color: orange;">${p.file.name}</a>`;
    } else {
      fileName = dv.fileLink(p.file.path);
    }
    
    return [
      fileName,
      new Date(p.file.ctime).toISOString().split('T')[0],
      new Date(p.file.mtime).toISOString().split('T')[0],
      progress
    ];
  })
);


```
