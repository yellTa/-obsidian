# 회사조사
 ```dataviewjs
// 데이터를 누적할 배열을 선언
let rows = [];

// 페이지를 순회하며 데이터를 배열에 추가
for (let page of dv.pages('"1_Projects/제로베이스 대기업 취업 특별반/멘토링/회사랑 자소서 조사/회사조사"').where(p => p.file.name != "회사조사")) {
    let createdDate = new Date(page.file.ctime);
    let modifiedDate = new Date(page.file.mtime);
    
    // 한 페이지에 대한 데이터를 배열에 추가
    rows.push([
        page.file.link,  // 파일 이름을 클릭 가능한 링크로 변환
        createdDate.toLocaleDateString("en-CA"),  // 'yyyy-MM-dd' 형식으로 출력
        modifiedDate.toLocaleDateString("en-CA"),  // 'yyyy-MM-dd' 형식으로 출력
        page.test ? `<span style="color:white; background-color:green; padding:2px; border-radius:4px;">True</span>` : page.코테
    ]);
}

// 누적된 데이터를 한 번에 테이블로 출력
dv.table(["제목", "생성 날짜", "수정 날짜", "코테"], rows);

```


# 백엔드 자소서
```dataview 
table 
dateformat(file.ctime, "yyyy-MM-dd") as "생성 날짜", 
dateformat(file.mtime, "yyyy-MM-dd") as "수정 날짜" 

FROM "1_Projects/제로베이스 대기업 취업 특별반/멘토링/회사랑 자소서 조사/백엔드자소서"
WHERE file.name != "백엔드자소서"
```
